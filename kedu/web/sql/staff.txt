CREATE TABLE staff (
	stfid       VARCHAR2(30)  NOT NULL, -- 사원id
	pwd         VARCHAR2(30)   NULL,     -- 비밀번호
	ismgr       CHAR(1)       NULL,     -- 사이트관리자YN
	stfnm       VARCHAR2(30)  NOT NULL, -- 이름
	jumin       VARCHAR2(13)  NOT NULL, -- 주민번호
	phone       VARCHAR2(20)  NULL,     -- 전화번호
	entrydt     DATE          NOT NULL, -- 입사일
	resignDt    DATE          NULL,     -- 퇴사일
	salary      NUMBER        NOT NULL, -- 급여
	pic_img     VARCHAR2(500) NULL,     -- 사진
	pic_img_orn VARCHAR2(500) NULL,     -- 사진주소
	address     VARCHAR2(300)  NULL,     -- 주소
	add_detail  VARCHAR2(300)  NULL,     -- 상세주소
	regdt       DATE          NOT NULL, -- 등록일
	regnm       VARCHAR(30)     NULL,     -- 등록자
	jobid      number       NULL,     -- 직급id
	deptid      NUMBER       NULL      -- 부서id
);
